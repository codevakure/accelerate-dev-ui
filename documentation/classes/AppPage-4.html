import { browser, by, element, until, protractor } from 'protractor';

export class AppPage {

  public navigateTo() {
    return  browser.get('/');
  }

  public enterUsername(text: string) {
    return element(by.id('username'))
      .sendKeys(text);
  }

  public enterPassword(text: string) {
    return element(by.id('password'))
      .sendKeys(text);
  }

  public clickLogin() {
    return element(by.id('login')).click();
  }
  public searchText() {
    return until.elementsLocated(by.className('profilename'));

  public selectDropdownbyNum = function ( elementID, optionNum ) {
    const field = element(by.id(elementID));
    if (optionNum){
      var options = field.all(by.tagName('option'))   
        .then(function(options){
          options[optionNum].click();
        });
    }
  }

  //use method to select elements from dropdown
  public selectOptionByOptionIndex(selectFormFieldElementId, index) {

    const formField = element(by.id(selectFormFieldElementId));
    formField.click().then(() => {
  
      formField.element(by.tagName('option'))
        .getAttribute('aria-owns').then((optionIdsString: string) => {
          const optionIds = optionIdsString.split(' ');
  
          const optionId = optionIds[index];
          const option = element(by.id(optionId));
          option.click();
        });
    });
  }
