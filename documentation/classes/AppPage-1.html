import { browser, by, element, until } from 'protractor';

export class AppPage {

  public navigateTo() {
    return  browser.get('/');
  }

  public enterUsername(text: string) {
    return element(by.id('username')).sendKeys(text);
  }

  //Script to add data to input or text box
  public enterinput(id,text){
    element(by.id(id)).sendKeys(text);
  }

  public enterPassword(text: string) {
    return element(by.id('password'))
      .sendKeys(text);
  }

  public clickLogin() {
    return element(by.id('login')).click();
  }
  public searchText() {
    return until.elementsLocated(by.className('profilename'));
  }
  public getSearchResultItems() {
    const condition = until.elementsLocated(by.className('modal-dialog'));
    return browser.wait(condition, 5000);
  }

  public igcePageLoaded(){
    const condition = until.elementsLocated(by.className('igceTitle'));
    return browser.wait(condition, 5000);
  } 

  public getTextField() {
    // const condition = until.elementsLocated(by.id('projecttitle')); 
    const elem = element(by.id('projecttitle')); 
    return elem;
  }

  public scrollTo(scrollToElement) {
    var wd = browser.driver;
    return scrollToElement.getLocation().then(function (loc) {
        return wd.executeScript('window.scrollTo(0,arguments[0]);', loc.y);
    });
  }
