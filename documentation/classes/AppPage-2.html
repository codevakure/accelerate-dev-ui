import { browser, by, element, until } from 'protractor';

export class AppPage {

  public navigateTo() {
    return  browser.get('/');
  }

  public enterUsername(text: string) {
    return element(by.id('username'))
      .sendKeys(text);
  }

  public enterPassword(text: string) {
    return element(by.id('password'))
      .sendKeys(text);
  }

  public clickLogin() {
    return element(by.id('login')).click();
  }
  public searchText() {
    return until.elementsLocated(by.className('profilename'));
  }
  public getSearchResultItems() {
    const condition = until.elementsLocated(by.className('modal-dialog'));
    return browser.wait(condition, 5000);
  }

  public igcePageLoaded(){
    // const condition = until.elementsLocated(by.className('igceTitle'));
    const condition = until.elementsLocated(by.id('general'));
    return browser.wait(condition, 5000);
  }

  public selectDropdownbyNum = function ( elementID, optionNum ) {
    const field = element(by.id(elementID));
    if (optionNum){
      var options = field.all(by.tagName('option'))   
        .then(function(options){
          options[optionNum].click();
        });
    }
  }
