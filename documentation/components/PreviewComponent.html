  scrollFunction() {
    if (
      document.body.scrollTop > 20 ||
      document.documentElement.scrollTop > 20

    } else {
      document.getElementById("myBtn").style.display = "none";
    }
  }

    document.body.scrollTop = 0; // For Safari
